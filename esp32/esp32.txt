// ---------------- TFT ----------------
#define TFT_CS   5

WiFiClientSecure net;

// ---------------- Control ----------------
bool messageSent = false;

// ---------------- Time (Sri Lanka local) ----------------
void syncTime() {
  // Sri Lanka = UTC+05:30
  configTime(5 * 3600, 30 * 60, "pool.ntp.org", "time.nist.gov");
  Serial.print("Syncing Sri Lanka time");

  time_t now = time(nullptr);
  unsigned long startMs = millis();

  while (now < 100000 && millis() - startMs < 20000) {
    delay(500);
    Serial.print(".");
    now = time(nullptr);
  }

  Serial.println();

  if (now >= 100000) {
    Serial.println("Time synced (Sri Lanka)");
  } else {
    Serial.println("Time sync timeout");
  }
}

String getTimestamp() {
  time_t now = time(nullptr);
  if (now < 100000) {
    return "1970-01-01T00:00:00";
  }

  struct tm* timeinfo = localtime(&now);
  char buffer[30];

  // Sri Lanka local time, no 'Z'
  strftime(buffer, sizeof(buffer), "%Y-%m-%dT%H:%M:%S", timeinfo);
  return String(buffer);
}

String getDisplayTime() {
  time_t now = time(nullptr);
  if (now < 100000) {
    return "Time Error";
  }

  struct tm* timeinfo = localtime(&now);
  char buffer[40];

  strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", timeinfo);
  return String(buffer);
}

void showStatus(const String& line1, const String& line2 = "", const String& line3 = "") {
  tft.fillScreen(ST77XX_BLACK);
  tft.setTextColor(ST77XX_WHITE);
  tft.setTextSize(1);
  tft.setCursor(0, 5);
  tft.println(line1);
  if (line2.length() > 0) tft.println(line2);
  if (line3.length() > 0) tft.println(line3);
}

void showReadings(int gasValue, uint16_t r, uint16_t g, uint16_t b, float temperature, float colorValue, const String& publishState, const String& displayTime) {
  tft.fillScreen(ST77XX_BLACK);
  tft.setTextColor(ST77XX_WHITE);
  tft.setTextSize(1);
  tft.setCursor(0, 0);

  tft.print("MQ: ");
  tft.println(gasValue);

  tft.print("R:");
  tft.print(r);
  tft.print(" G:");
  tft.print(g);
  tft.print(" B:");
  tft.println(b);

  tft.print("Tmp: ");
  if (temperature == DEVICE_DISCONNECTED_C) {
    tft.println("Err");
  } else {
    tft.print(temperature, 1);
    tft.println(" C");
  }

  tft.print("Col: ");
  tft.print(colorValue, 1);
  tft.println("%");

  tft.print("AWS: ");
  tft.println(publishState);

  tft.println(displayTime);
}

void connectWiFi() {
  showStatus("Connecting WiFi...");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Connecting to WiFi");

  unsigned long startMs = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - startMs < 20000) {
    delay(500);
    Serial.print(".");
  }

  Serial.println();

  if (WiFi.status() == WL_CONNECTED) {
    Serial.print("WiFi connected. ESP32 IP: ");
    Serial.println(WiFi.localIP());
    showStatus("WiFi Connected", WiFi.localIP().toString());
  } else {
    Serial.println("WiFi connection timeout");
    showStatus("WiFi Failed");
  }
}

bool connectAWS() {
  showStatus("Connecting AWS...");
  net.setCACert(AWS_CERT_CA);
  net.setCertificate(AWS_CERT_CRT);
  net.setPrivateKey(AWS_CERT_PRIVATE);

  client.setServer(AWS_IOT_ENDPOINT, AWS_IOT_PORT);

  Serial.print("Connecting to AWS IoT");

  unsigned long startMs = millis();
  while (!client.connected() && millis() - startMs < 20000) {
    if (client.connect(CLIENT_ID)) {
      Serial.println();
      Serial.println("Connected to AWS IoT");
      showStatus("AWS Connected");
      return true;
    } else {
      Serial.print(".");
      delay(2000);
    }
  }

  Serial.println();
  Serial.println("AWS connection timeout/failed");
  showStatus("AWS Failed");
  return false;
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  tft.initR(INITR_BLACKTAB);   // if white screen, try INITR_GREENTAB or INITR_REDTAB
  tft.setRotation(1);
  tft.setTextWrap(true);

  showStatus("Starting test...");

  if (!tcs.begin()) {
    Serial.println("TCS34725 not found");
    showStatus("TCS34725 Error");
    while (1);
  }

  tempSensor.begin();

  connectWiFi();
  syncTime();
  connectAWS();
}

void loop() {
  if (messageSent) {
    client.loop();
    return;
  }

  client.loop();

  // Read MQ sensor
  int gasValue = analogRead(mqPin);

  // Read color sensor
  uint16_t r, g, b, c;
  tcs.getRawData(&r, &g, &b, &c);

  // Read temperature sensor
  tempSensor.requestTemperatures();
  float temperature = tempSensor.getTempCByIndex(0);

  // Build numeric COLOR value
  float colorValue = 0.0;
  uint32_t total = (uint32_t)r + (uint32_t)g + (uint32_t)b;
  if (total > 0) {
    colorValue = ((float)r / (float)total) * 100.0f;
  }

  // Sri Lanka timestamp
  String timestamp = getTimestamp();
  String displayTime = getDisplayTime();

  // Build JSON in required format
  StaticJsonDocument<256> doc;
  doc["DEVICE_ID"] = "DEV001";
  doc["TIMESTAMP"] = timestamp;
  doc["TYPE"] = "SENSOR";
  doc["FACTORY_ID"] = "FAC001";
  doc["BATCH_ID"] = "BAT001";
  doc["COLOR"] = colorValue;
  doc["TEMPERATURE"] = temperature;
  doc["MQ135"] = gasValue;

  char payload[256];
  serializeJson(doc, payload);

  // Serial output
  Serial.println("Prepared JSON:");
  Serial.println(payload);
  Serial.print("Gas: ");
  Serial.print(gasValue);
  Serial.print(" | R: ");
  Serial.print(r);
  Serial.print(" G: ");
  Serial.print(g);
  Serial.print(" B: ");
  Serial.print(b);
  Serial.print(" | Temp: ");
  Serial.print(temperature);
  Serial.print(" C");
  Serial.print(" | ColorValue: ");
  Serial.print(colorValue, 1);
  Serial.print(" | Time: ");
  Serial.println(displayTime);

  // Publish once
  bool ok = false;
  if (client.connected()) {
    ok = client.publish(MQTT_TOPIC, payload);
  }

  if (ok) {
    Serial.println("Publish status: SUCCESS");
    showReadings(gasValue, r, g, b, temperature, colorValue, "SUCCESS", displayTime);
  } else {
    Serial.println("Publish status: FAILED");
    showReadings(gasValue, r, g, b, temperature, colorValue, "FAILED", displayTime);
  }

  messageSent = true;
}
